\section{Hardware/Software}\label{hardwaresoftware}

\textbf{Front-End Software}

\begin{longtable}{|p{0.22\textwidth}|p{0.70\textwidth}|}
\caption{Frond-End Software Requirements for Development}
\hline
\centering{\textbf{Software}} & \centering{\textbf{Description}} \\
\hline
Figma & This browser-based collaborative tool is used to make
interactive prototypes, wireframes, and vector-based interface designs
for websites and mobile applications. \\
\hline
Blender & A free and open-source 3D creation suite used for 3D modeling,
sculpting, animation, visual effects, rendering, and video editing. Its
main purpose is to make 3D assets. \\
\hline
Unity & This game engine is used to create 2D, 3D, augmented reality
(AR), and virtual reality (VR) games and interactive simulations,
providing a platform for all game-making needs. \\
\hline
Visual Studio & A strong editing and debugging environment for front-end
code, especially when part of full-stack applications built on the
Microsoft platform. \\
\hline
Photoshop & The standard raster graphics editor in the industry, mainly
used for creating UI elements within the game. \\
\hline
FL Studio & A comprehensive Digital Audio Workstation (DAW) used for
making sound effects and background music within the game. \\
\hline
\end{longtable}

\textbf{Back-End Software}

\begin{longtable}{|p{0.22\textwidth}|p{0.70\textwidth}|}
\caption{Back-End Software Requirements}
\hline
\centering{\textbf{Software}} & \centering{\textbf{Description}} \\
\hline

Git & A distributed version control system used for tracking code changes

\hline
GitHub & A web-based hosting service for Git repositories, providing
collaboration tools and code review capabilities. \\

\end{longtable}

\textbf{Hardware Requirements (Development)}

\begin{longtable}{|p{0.22\textwidth}|p{0.70\textwidth}|}
\caption{Hardware Requirements for Development}
\hline
\centering{\textbf{Hardware}} & \centering{\textbf{Description}} \\
\hline
Processor: Intel i5 or higher & A mid-range CPU capable of efficiently
running instructions and handling multitasking. \\
\hline
RAM: 8GB or higher & Short-term memory used by active programs; 8GB is
the baseline for modern development work. \\
\hline
GPU: GeForce 400 with 2GB VRAM or higher & A dedicated graphics card
needed for rendering 3D graphics in Blender and Unity. \\
\hline
Storage: 240GB SSD or higher & Fast flash-based storage enabling quick
boot and load times. Minimum capacity: 240GB. \\
\hline
\end{longtable}
